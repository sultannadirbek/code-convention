Пример 1: Создание нового заказа

Этот Use Case отвечает за создание нового заказа. Он валидирует данные пользователя и товаров, создаёт заказ и отправляет уведомление.

Шаг 1: Интерфейс Use Case


interface CreateOrderUseCase
{
    public function execute(CreateOrderInput $input): CreateOrderOutput;
}

Шаг 2: Input и Output объекты

Эти объекты используются для передачи данных в Use Case и получения результатов выполнения.


class CreateOrderInput
{
    public int $userId;
    public array $products; // Массив товаров
    public string $shippingAddress;

    public function __construct(int $userId, array $products, string $shippingAddress)
    {
        $this->userId = $userId;
        $this->products = $products;
        $this->shippingAddress = $shippingAddress;
    }
}

class CreateOrderOutput
{
    public bool $success;
    public ?string $message;
    public ?Order $order;

    public function __construct(bool $success, ?string $message = null, ?Order $order = null)
    {
        $this->success = $success;
        $this->message = $message;
        $this->order = $order;
    }
}

Шаг 3: Реализация Use Case


class CreateOrderUseCase implements CreateOrderUseCase
{
    private UserService $userService;
    private ProductService $productService;
    private OrderService $orderService;
    private NotificationService $notificationService;

    public function __construct(
        UserService $userService,
        ProductService $productService,
        OrderService $orderService,
        NotificationService $notificationService
    ) {
        $this->userService = $userService;
        $this->productService = $productService;
        $this->orderService = $orderService;
        $this->notificationService = $notificationService;
    }

    public function execute(CreateOrderInput $input): CreateOrderOutput
    {
        // Шаг 1: Валидация пользователя
        $user = $this->userService->findById($input->userId);
        if (!$user) {
            return new CreateOrderOutput(false, 'Пользователь не найден');
        }

        // Шаг 2: Валидация товаров
        $products = [];
        foreach ($input->products as $productId) {
            $product = $this->productService->findById($productId);
            if (!$product) {
                return new CreateOrderOutput(false, "Товар с ID $productId не найден");
            }
            $products[] = $product;
        }

        // Шаг 3: Создание заказа
        $order = new Order($user, $products, $input->shippingAddress);
        $this->orderService->save($order);

        // Шаг 4: Отправка уведомления пользователю
        $this->notificationService->sendOrderCreatedNotification($user, $order);

        // Возврат успешного ответа
        return new CreateOrderOutput(true, 'Заказ успешно создан', $order);
    }
}

Пример 2: Изменение статуса заказа

Этот Use Case изменяет статус существующего заказа.

Input и Output объекты


class ChangeOrderStatusInput
{
    public int $orderId;
    public string $newStatus;

    public function __construct(int $orderId, string $newStatus)
    {
        $this->orderId = $orderId;
        $this->newStatus = $newStatus;
    }
}

class ChangeOrderStatusOutput
{
    public bool $success;
    public ?string $message;

    public function __construct(bool $success, ?string $message = null)
    {
        $this->success = $success;
        $this->message = $message;
    }
}

Use Case для изменения статуса заказа


class ChangeOrderStatusUseCase implements ChangeOrderStatusUseCase
{
    private OrderService $orderService;

    public function __construct(OrderService $orderService)
    {
        $this->orderService = $orderService;
    }

    public function execute(ChangeOrderStatusInput $input): ChangeOrderStatusOutput
    {
        // Шаг 1: Найти заказ
        $order = $this->orderService->findById($input->orderId);
        if (!$order) {
            return new ChangeOrderStatusOutput(false, 'Заказ не найден');
        }

        // Шаг 2: Изменить статус заказа
        $order->setStatus($input->newStatus);
        $this->orderService->save($order);

        // Возврат успешного ответа
        return new ChangeOrderStatusOutput(true, 'Статус заказа успешно изменён');
    }
}
